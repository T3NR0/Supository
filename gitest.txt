git init	||	pour commencer la vie
git status	||	pour connaitre l'état du dossier
git add		||	pour ajouter un fichier
git commit	||	pour commiter sa maman
	-m		"y ajouter un message"
	-a		commit tous les fichiers addés dans le passé
git log		||	pour connaitre l'histore de la vie
git checkout	||	pour se positionner sur un SHA (un gros shashoo)
	master		pour retourner dans la vraie vie
